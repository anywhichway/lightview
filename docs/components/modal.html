Modal

Modal is used to show a dialog or a box when you click a button. Uses the native HTML <dialog> element.

Quick Example

Hello!

This is a simple modal dialog.

Usage

import Modal from './components/actions/modal.js';
import Button from './components/actions/button.js';

// Open modal programmatically
Modal.open('my_modal');

// Or with the component
Modal({ id: 'my_modal' },
    Modal.Box({},
        tags.h3({ class: 'text-lg font-bold' }, 'Hello!'),
        tags.p({ class: 'py-4' }, 'This is a modal dialog.'),
        Modal.Action({},
            Button({ onclick: () => Modal.close('my_modal') }, 'Close')
        )
    ),
    Modal.Backdrop({})
)
