Navbar

Navbar is a horizontal navigation bar at the top of the page. Supports start, center, and end sections.

Basic Example

Navbar with brand and menu items

const { default: Navbar } = await import('/components/layout/navbar.js');
const { default: Button } = await import('/components/actions/button.js');
const { default: Menu } = await import('/components/navigation/menu.js');

const { tags, $ } = Lightview;

const navbar = Navbar({ class: 'bg-base-100 rounded-box' },
    Navbar.Start({},
        Button({ ghost: true, class: 'text-xl' }, 'ðŸš€ Lightview')
    ),
    Navbar.Center({ class: 'hidden lg:flex' },
        Menu({ horizontal: true },
            Menu.Item({}, Menu.Link({}, 'Features')),
            Menu.Item({}, Menu.Link({}, 'Pricing')),
            Menu.Item({}, Menu.Link({}, 'About'))
        )
    ),
    Navbar.End({},
        Button({ color: 'primary' }, 'Get Started')
    )
);

$('#example').content(navbar);

const { default: Navbar } = await import('/components/layout/navbar.js');
const { default: Button } = await import('/components/actions/button.js');
const { default: Menu } = await import('/components/navigation/menu.js');
const { tags, $ } = Lightview;
tags.Navbar = Navbar;
tags.Button = Button;
tags.Menu = Menu;
tags['Navbar.Start'] = Navbar.Start;
tags['Navbar.Center'] = Navbar.Center;
tags['Navbar.End'] = Navbar.End;
tags['Menu.Item'] = Menu.Item;
tags['Menu.Link'] = Menu.Link;

const navbar = {
    Navbar: {
        class: 'bg-base-100 rounded-box',
        children: [
            {
                'Navbar.Start': {
                    children: [{ Button: { ghost: true, class: 'text-xl', text: 'ðŸš€ Lightview' } }]
                }
            },
            {
                'Navbar.Center': {
                    class: 'hidden lg:flex',
                    children: [
                        {
                            Menu: {
                                horizontal: true,
                                children: [
                                    { 'Menu.Item': { children: [{ 'Menu.Link': { text: 'Features' } }] } },
                                    { 'Menu.Item': { children: [{ 'Menu.Link': { text: 'Pricing' } }] } },
                                    { 'Menu.Item': { children: [{ 'Menu.Link': { text: 'About' } }] } }
                                ]
                            }
                        }
                    ]
                }
            },
            {
                'Navbar.End': {
                    children: [{ Button: { color: 'primary', text: 'Get Started' } }]
                }
            }
        ]
    }
};

$('#example').content(navbar);
