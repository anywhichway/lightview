Tabs

Tabs allow showing and hiding content to organize information.

Quick Example

Tab 1 Tab 2 Tab 3 
import Tabs from './components/navigation/tabs.js';

// Using Lightview component
Tabs({ variant: 'bordered' },
    Tabs.Tab({}, 'Tab 1'),
    Tabs.Tab({ active: true }, 'Tab 2'),
    Tabs.Tab({}, 'Tab 3')
)

// Or using DaisyUI classes directly
<div role="tablist" class="tabs tabs-bordered">
  <a role="tab" class="tab">Tab 1</a>
  <a role="tab" class="tab tab-active">Tab 2</a>
  <a role="tab" class="tab">Tab 3</a>
</div>

Reactive Example

const { signal, tags } = Lightview;
const { div } = tags;
const Tabs = LightviewX.getComponent('Tabs');

const activeTab = signal(0);
const tabs = ['Profile', 'Settings', 'Messages'];

const demo = div({ class: 'flex flex-col gap-4' },
    Tabs({ variant: 'bordered' },
        tabs.map((label, i) => 
            Tabs.Tab({ 
                active: () => activeTab.value === i,
                onclick: () => activeTab.value = i
            }, label)
        )
    ),
    div({ class: 'p-4 border rounded bg-base-100' },
        () => `Content for ${tabs[activeTab.value]}`
    )
);
For more examples and options, see the DaisyUI Tab documentation.
